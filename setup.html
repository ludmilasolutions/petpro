rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // FUNCIONES DE AYUDA function isAuthenticated() { return request.auth != null; } function isOwner(petId) { return resource.data.dueñoId == request.auth.uid; } function isAuthorizedVet(petId) { return resource.data.veterinariasAutorizadas.hasAny([request.auth.uid]); } function isVeterinary() { return exists(/databases/$(database)/documents/veterinarias/$(request.auth.uid)); } function isAdmin() { return exists(/databases/$(database)/documents/admins/$(request.auth.uid)); } // DUEÑOS match /dueños/{ownerId} { allow create: if isAuthenticated() && request.auth.uid == ownerId; allow read, update, delete: if isAuthenticated() && request.auth.uid == ownerId; } // VETERINARIAS match /veterinarias/{vetId} { allow create: if isAuthenticated() && request.auth.uid == vetId; allow read, update: if isAuthenticated() && request.auth.uid == vetId; allow delete: if false; // No permitir eliminación por seguridad } // MASCOTAS match /mascotas/{petId} { allow create: if isAuthenticated(); allow read: if isAuthenticated() && (isOwner(petId) || isAuthorizedVet(petId)); allow update: if isAuthenticated() && (isOwner(petId) || isVeterinary()); allow delete: if isAuthenticated() && isOwner(petId); // Validaciones allow create: if request.resource.data.keys().hasAll(['nombre', 'especie', 'dueñoId']) && request.resource.data.dueñoId == request.auth.uid; } // HISTORIAL MÉDICO match /historial_medico/{recordId} { allow create: if isAuthenticated() && isVeterinary(); allow read: if isAuthenticated(); allow update, delete: if isAuthenticated() && isVeterinary(); // Auditoría obligatoria allow create: if request.resource.data.keys().hasAll(['mascotaId', 'tipo', 'usuarioId']) && request.resource.data.usuarioId == request.auth.uid; } // TURNOS match /turnos/{appointmentId} { allow create: if isAuthenticated(); allow read: if isAuthenticated(); allow update: if isAuthenticated() && ( request.auth.uid == resource.data.veterinariaId || request.auth.uid == resource.data.dueñoId ); allow delete: if isAuthenticated() && ( request.auth.uid == resource.data.veterinariaId || request.auth.uid == resource.data.dueñoId ); } // AUTORIZACIONES match /autorizaciones/{authId} { allow create: if isAuthenticated() && isVeterinary(); allow read: if isAuthenticated(); allow update: if isAuthenticated() && request.auth.uid == resource.data.dueñoId; allow delete: if isAuthenticated() && request.auth.uid == resource.data.dueñoId; } // PAGOS match /pagos/{paymentId} { allow create: if isAuthenticated() && isVeterinary(); allow read: if isAuthenticated() && request.auth.uid == resource.data.veterinariaId; allow update: if isAdmin(); } // REFUGIOS (públicos) match /refugios/{shelterId} { allow read: if true; allow create, update, delete: if isAdmin(); } // NOTIFICACIONES match /notificaciones/{notificationId} { allow create: if isAuthenticated(); allow read: if isAuthenticated() && request.auth.uid == resource.data.userId; allow update: if isAuthenticated() && request.auth.uid == resource.data.userId; } // CONFIGURACIÓN match /configuracion/{configId} { allow read: if true; allow write: if isAdmin(); } // USUARIOS INTERNOS DE VETERINARIA match /usuarios_internos/{userId} { allow create: if isAuthenticated() && isVeterinary(); allow read: if isAuthenticated() && isVeterinary(); allow update: if isAuthenticated() && isVeterinary(); allow delete: if isAuthenticated() && isVeterinary(); } } }
